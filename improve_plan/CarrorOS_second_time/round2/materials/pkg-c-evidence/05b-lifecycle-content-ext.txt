===== .omc/reference/SUBAGENT.md =====
     1	# SubAgent 派发契约
     2	
     3	> 参考 round3 设计：SubAgent 是轻量任务派发机制，仅在需要时才使用。
     4	
     5	## 何时派发
     6	- 局部调研（读代码/查文档）
     7	- 单一改动（≤5 文件、≤5 分钟）
     8	- 不熟悉的局部技术栈
     9	
    10	**不做**：把整个任务丢给 subagent。要自己做主流程，subagent 只做局部。
    11	
    12	## 派发四步
    13	
    14	1. **写输入**：`subagent/{id}_input.md`
    15	   - Goal（目标）
    16	   - Scope（允许读/改/禁止的范围）
    17	   - 期望产出（什么格式、什么内容）
    18	   - Stop 条件（什么情况下必须停止并返回）
    19	
    20	2. **spawn**：用 delegate_task 工具派发，传 id
    21	
    22	3. **轮询**：每 10s 读 `subagent/{id}.json` 的 status
    23	   - `done` → 合并 deliverable
    24	   - `blocked` → 补信息 / 拆小重派 / 自己接管
    25	   - `15s 无 ACK` 或 `5min 未 done` → 收 partial → 自己接管
    26	
    27	4. **收官**：按状态处理
    28	
    29	## 铁律约束
    30	SubAgent 同样受铁律约束：
    31	- 越界修改 → 违规
    32	- 无证据改动 → 违规
    33	- 埋头不报 > 15s → 违规
    34	
    35	## 契约格式
    36	
    37	```json
    38	{
    39	  "subagent_id": "research-01",
    40	  "goal": "调研 X 模块的 API 设计",
    41	  "scope": {
    42	    "allow_read": ["src/x/**"],
    43	    "allow_write": [],
    44	    "forbidden": ["secrets/**", ".env"]
    45	  },
    46	  "expected_output": "report.md 格式：背景、API 设计、风险点",
    47	  "stop_on": ["timeout_300s", "error_3_consecutive"]
    48	}
    49	```
    50	
    51	## 参考
    52	- poll-subagent.py — 轮询脚本
    53	- subagent-schema.json — 输入/输出 JSON 模式

     1	#!/usr/bin/env python3

     3	goal_state_machine.py — Goal 自闭环状态机

     8	  - intent/goal 缺失时自动回退 CLARIFY

    10	  - archive 成功 → ARCHIVED
    11	
    12	Usage:
    13	    from goal_state_machine import GoalMachine, GoalStatus
    14	
    15	Usage:
    16	    gm = GoalMachine(token_path)
    17	    gm.transition("VERIFYING")
    18	    print(gm.current_state)
    19	"""
    20	
    21	import json
    22	from datetime import datetime, timezone
    23	from pathlib import Path
    24	
    25	# ─── State Constants ───
    26	CLARIFY = "CLARIFY"
    27	PLANNING = "PLANNING"
    28	EXECUTING = "EXECUTING"
    29	VERIFYING = "VERIFYING"
    30	ARCHIVING = "ARCHIVING"
    31	ARCHIVED = "ARCHIVED"
    32	
    33	ALL_STATES = [CLARIFY, PLANNING, EXECUTING, VERIFYING, ARCHIVING, ARCHIVED]
    34	
    35	# ─── Valid Transitions ───
    36	_VALID_TRANSITIONS = {
    37	    None: [CLARIFY],                  # 初始状态 -> CLARIFY
    38	    CLARIFY: [PLANNING, CLARIFY],     # 澄清后可进 PLANNING，或继续澄清
    39	    PLANNING: [EXECUTING, CLARIFY],   # 计划后执行，或回 CLARIFY（需求变更）
    40	    EXECUTING: [VERIFYING, CLARIFY],  # 执行后验证，或回 CLARIFY
    41	    VERIFYING: [ARCHIVING, EXECUTING, CLARIFY],  # 验证后归档/回执行/回澄清
    42	    ARCHIVING: [ARCHIVED, VERIFYING], # 归档中 -> 完成或回验证
    43	    ARCHIVED: [],                     # 终态
    44	}
    45	
    46	
    47	class GoalError(Exception):
    48	    """GoalMachine 状态转换异常"""
    49	    pass
    50	
    51	
    52	class GoalMachine:
    53	    """Goal 状态机 — 管理任务生命周期状态转换"""
    54	
    55	    def __init__(self, token_path=None, spec_path=None):
    56	        self.token_path = Path(token_path) if token_path else None
    57	        self.spec_path = spec_path
    58	        self._state = None
    59	
    60	        # 尝试从 token 恢复状态
    61	        if self.token_path and self.token_path.exists():
    62	            try:
    63	                token = json.loads(self.token_path.read_text())
    64	                self._state = token.get("goal", {}).get("state")
    65	            except (json.JSONDecodeError, OSError):
    66	                pass
    67	
    68	        if self._state not in ALL_STATES:
    69	            self._state = None  # 还未初始化
    70	
    71	    @property
    72	    def current_state(self):
    73	        return self._state
    74	
    75	    @property
    76	    def is_terminal(self):
    77	        return self._state == ARCHIVED
    78	
    79	    def can_transition(self, target_state):
    80	        """检查 target_state 是否合法"""
    81	        return target_state in _VALID_TRANSITIONS.get(self._state, [])
    82	
    83	    def transition(self, target_state, token=None, reason=""):
    84	        """尝试状态转换 — 验证合法性 + 更新 token（如有）"""
    85	        if target_state not in ALL_STATES:
    86	            raise GoalError(f"Unknown state: {target_state}")
    87	
    88	        valid = _VALID_TRANSITIONS.get(self._state, [])
    89	        if target_state not in valid:
    90	            raise GoalError(
    91	                f"Invalid transition: {self._state} → {target_state} "
    92	                f"(allowed: {valid})"
    93	            )
    94	
    95	        old_state = self._state
    96	        self._state = target_state
    97	
    98	        # 更新 token
    99	        token_data = token
   100	        if token_data is None and self.token_path and self.token_path.exists():
   101	            try:
   102	                token_data = json.loads(self.token_path.read_text())
   103	            except (json.JSONDecodeError, OSError):
   104	                pass
   105	
   106	        if token_data is not None:
   107	            if "goal" not in token_data:
   108	                token_data["goal"] = {}
   109	            token_data["goal"]["state"] = target_state
   110	            token_data["goal"]["previous_state"] = old_state
   111	            token_data["goal"]["transitions"] = token_data["goal"].get("transitions", 0) + 1
   112	            token_data["goal"]["last_transition"] = datetime.now(timezone.utc).isoformat()
   113	            if reason:
   114	                token_data["goal"]["last_reason"] = reason
   115	            if self.token_path:
   116	                self.token_path.parent.mkdir(parents=True, exist_ok=True)
   117	                self.token_path.write_text(
   118	                    json.dumps(token_data, indent=2, ensure_ascii=False) + "\n"
   119	                )
   120	
   121	        return True
   122	
   123	    def auto_progress(self, token=None):
   124	        """根据 token 状态自动推进（executing→verifying→archiving→archived）"""
   125	        token_data = token
   126	        if token_data is None and self.token_path and self.token_path.exists():
   127	            try:
   128	                token_data = json.loads(self.token_path.read_text())
   129	            except (json.JSONDecodeError, OSError):
   130	                return []
   131	
   132	        if not token_data:
   133	            return []
   134	
   135	        stats = token_data.get("stats", {})
   136	        done = stats.get("done", 0)
   137	        total = stats.get("total", 0)
   138	        goal_state = token_data.get("goal", {}).get("state")
   139	
   140	        transitions_made = []
   141	
   142	        if goal_state == EXECUTING and done >= total:
   143	            self.transition(VERIFYING, token_data,
   144	                            reason=f"auto: all {done}/{total} steps completed")
   145	            transitions_made.append(("auto", EXECUTING, VERIFYING))
   146	
   147	        if goal_state == VERIFYING and done >= total:
   148	            self.transition(ARCHIVING, token_data,
   149	                            reason="auto: all steps verified")
   150	            transitions_made.append(("auto", VERIFYING, ARCHIVING))
   151	
   152	        # ARCHIVING → ARCHIVED 需要外部调用 archive 命令后自动触发
   153	        return transitions_made
   154	
   155	    def reset(self, token=None):
   156	        """重置状态机 — 回到 CLARIFY"""
   157	        self._state = None
   158	        return self.transition(CLARIFY, token, reason="reset")
   159	
   160	    def get_summary(self):
   161	        """获取状态机摘要"""
   162	        return {
   163	            "current_state": self._state,
   164	            "is_terminal": self.is_terminal,
   165	            "valid_transitions": _VALID_TRANSITIONS.get(self._state, []),
   166	        }
   167

     1	#!/usr/bin/env python3
     2	"""
     3	handoff_writer.py — Resume Capsule 生成器
     4	
     5	生成 handoff.md（导航-only，非真相源）。

    15	def generate_capsule(
    16	    task_id: str,
    17	    token: dict,
    18	    plan_path: Optional[Path] = None,
    19	    executor_path: Optional[Path] = None,
    20	    max_nav_chars: int = 8000,
    21	    archived: bool = False,
    22	) -> str:
    23	    """
    24	    生成 Resume Capsule（导航-only）。
    25	    头部强制 NOT_SOURCE_OF_TRUTH。
    26	    如果 archived=True，标记已归档，无 next_action。
    27	    """
    28	    lines = []
    29	    lines.append("# Resume Capsule — Navigation Only")
    30	    lines.append("")
    31	    lines.append("## ⚠️ NOT SOURCE OF TRUTH")
    32	    lines.append("Resume engine MUST load token.json (CAS) first.")

    52	            stripped = line.strip()
    53	            if stripped.startswith("-") or stripped.startswith("*"):

    57	    lines.append("")
    58	
    59	    # Next action (or archived)

    63	        lines.append("  For reference only. Final report in final-report.md.")
    64	    else:
    65	        lines.append("## Next Action")
    66	        cs = token.get("task", {}).get("current_step", "?")
    67	        st = token.get("task", {}).get("status", "?")

    70	
    71	    # Changed files from plan scope
    72	    lines.append("## Changed Files")
    73	    if plan_path and plan_path.exists():
    74	        text = plan_path.read_text(encoding="utf-8")

    86	    lines.append("")

    90	    lines.append("  - token.json (CAS)")
    91	    if plan_path:
    92	        lines.append(f"  - {plan_path.relative_to(plan_path.parents[3])}")
    93	    lines.append("")
    94	
    95	    # Do not reload
    96	    lines.append("## Do Not Reload")
    97	    lines.append("  - 全部旧 transcript")
    98	    lines.append("  - docs/reviews/**")
    99	    lines.append("  - 完整测试日志")
   100	    lines.append("  - 旧 Oracle 裁决")
   101	
   102	    text = "\n".join(lines)
   103	    if len(text) > max_nav_chars:
   104	        text = text[: max_nav_chars - 20] + "\n...[TRUNCATED]"
   105	    return text

   108	def write_handoff(
   109	    task_dir: Path,
   110	    task_id: str,
   111	    token: dict,
   112	    plan_path: Optional[Path] = None,
   113	    executor_path: Optional[Path] = None,
   114	    archived: bool = False,
   115	) -> Path:
   116	    """Write handoff.md to task directory."""
   117	    capsule = generate_capsule(task_id, token, plan_path, executor_path, archived=archived)
   118	    handoff_path = task_dir / "handoff.md"
   119	    handoff_path.write_text(capsule, encoding="utf-8")
   120	    return handoff_path

   122	
   123	def run_preflight(task_dir: Path, token: dict) -> list[str]:
   124	    """
   125	    Resume Preflight 检查：
   126	    1) token.json CAS load
   127	    2) plan 版本一致性
   128	    3) external_effects 三界检查
   129	    4) handoff 存在
   130	
   131	    Returns list of issues (empty = pass).
   132	    """
   133	    issues = []
   134	
   135	    # 1. token.json CAS check
   136	    if not token.get("schema_version"):
   137	        issues.append("⚠ token.json missing schema_version (CAS)")
   138	    if not token.get("stats"):
   139	        issues.append("⚠ token.json missing stats section")
   140	
   141	    # 2. plan consistency
   142	    plan_path = task_dir / "plan.md"
   143	    if not plan_path.exists():
   144	        issues.append("⚠ plan.md not found")
   145	    else:
   146	        plan_text = plan_path.read_text(encoding="utf-8")
   147	        if not plan_text.strip():
   148	            issues.append("⚠ plan.md is empty")
   149	
   150	    # 3. external_effects — check state/sub_task dirs for IN_FLIGHT markers
   151	    state_dir = task_dir / "state"
   152	    if state_dir.exists():
   153	        for f in state_dir.iterdir():
   154	            if f.suffix == ".json":
   155	                try:
   156	                    data = json.loads(f.read_text(encoding="utf-8"))
   157	                    effects = data.get("external_effects", []) if isinstance(data, dict) else []
   158	                    for eff in effects:
   159	                        status = eff.get("status", "")
   160	                        if status in ("IN_FLIGHT", "UNKNOWN"):
   161	                            issues.append(f"⚠ external_effect IN_FLIGHT/UNKNOWN: {eff.get('id', '?')}")
   162	                except (json.JSONDecodeError, OSError):
   163	                    pass
   164	
   165	    # 4. handoff exists
   166	    handoff_path = task_dir / "handoff.md"
   167	    if not handoff_path.exists():
   168	        issues.append("ℹ handoff.md not yet written (acceptable for in-progress)")
   169	    else:
   170	        text = handoff_path.read_text(encoding="utf-8")
   171	        if "NOT SOURCE OF TRUTH" not in text:
   172	            issues.append("⚠ handoff.md missing NOT_SOURCE_OF_TRUTH header")
   173	
   174	    return issues
