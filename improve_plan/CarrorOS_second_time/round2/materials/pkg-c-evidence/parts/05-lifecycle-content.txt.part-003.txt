   780	def main():
   781	    if len(sys.argv) < 2 or sys.argv[1] not in KNOWN_SUBCOMMANDS:
   782	        # 无参数或参数不是已知子命令 → 当作目标描述自动激活
   783	        goal = " ".join(sys.argv[1:]) if len(sys.argv) > 1 else ""
   784	        if goal:
   785	            cmd_on(goal)
   786	        else:
   787	            cmd_status()
   788	        return
   789	
   790	    cmd_name = sys.argv[1]
   791	    args = sys.argv[2:]
   792	
   793	    if cmd_name == "on":
   794	        goal = args[0] if args else "目标任务未指定"
   795	        expiry = int(args[1]) if len(args) > 1 else 6
   796	        cmd_on(goal, expiry)
   797	    elif cmd_name == "set":
   798	        if len(args) < 2:
   799	            print("用法: lx-goal.py set <key> <value>")
   800	            sys.exit(1)
   801	        cmd_set(args[0], args[1])
   802	    elif cmd_name == "subagent-log":
   803	        if not args:
   804	            print("用法: lx-goal.py subagent-log assign|complete|fail|summary <agent> <subtask> [detail]")
   805	            sys.exit(1)
   806	        cmd_subagent_log(*args[:4])
   807	    elif cmd_name == "_update-lock":
   808	        # kwargs parsing: key=value
   809	        kwargs = {}
   810	        for arg in args:
   811	            if "=" in arg:
   812	                k, v = arg.split("=", 1)
   813	                kwargs[k] = v
   814	        cmd_update_lock(**kwargs)
   815	    elif cmd_name == "skip-risk":
   816	        # skip-risk "描述" [risk_level] [reason] [impact]
   817	        cmd_skip_risk(*(args[:4]))
   818	    elif cmd_name in ("task-done", "hard-boundary-hit", "blocked-human"):
   819	        handlers = {
   820	            "task-done": cmd_task_done,
   821	            "hard-boundary-hit": cmd_hard_boundary_hit,
   822	            "blocked-human": cmd_blocked_human,
   823	        }
   824	        handlers[cmd_name](*(args[:3]))
   825	    else:
   826	        # off, status, phase0-done, report, poll, retry, done — 无参数
   827	        rc = KNOWN_SUBCOMMANDS[cmd_name]()
   828	        if isinstance(rc, int):
   829	            sys.exit(rc)
   830

   832	if __name__ == "__main__":
   833	    main()
