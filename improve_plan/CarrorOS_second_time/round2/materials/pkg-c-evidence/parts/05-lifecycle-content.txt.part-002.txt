   590	    """记录跳过的风险。
   591	
   592	    risk_level: low/medium/high/critical。

   594	    反馈给人类干预 — 这是 goal 模式的核心安全阀。
   595	    """
   596	    risk_level = (risk_level or "low").lower()
   597	    if risk_level not in ("low", "medium", "high", "critical"):
   598	        risk_level = "low"
   599	    mode_data, path = _read_mode_file()
   600	    mode_data.setdefault("skipped_risks", []).append({
   601	        "description": description,
   602	        "risk_level": risk_level,
   603	        "reason": reason,
   604	        "impact": impact,
   605	        "timestamp": get_now(),
   606	    })
   607	    _write_mode_file(mode_data, path)
   608	
   609	    plan_dir = _get_plan_dir(mode_data)
   610	    if plan_dir:
   611	        plan_md = plan_dir / "plan.md"
   612	        if plan_md.exists():
   613	            with open(plan_md, "a", encoding="utf-8") as f:
   614	                f.write(f"\n- [skip-risk/{risk_level}] {description} — {reason or '未填理由'}  ({get_now()})\n")
   615	        _update_lock_counter(plan_dir, "skipped_risks")
   616	
   617	    marker = "⚠️" if risk_level in ("medium", "high", "critical") else "📝"
   618	    print(f"{marker} 已记录跳过的风险[{risk_level}]: {_sanitize(description)}")
   619	    if risk_level in ("medium", "high", "critical"):
   620	        print("   中高风险项：仅跳过不执行，将出现在退出报告「需人为决策汇总」")
