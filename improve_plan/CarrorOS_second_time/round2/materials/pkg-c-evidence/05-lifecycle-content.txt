     1	#!/usr/bin/env python3
     2	"""
     3	session-start.py — CarrorOS SessionStart hook（compact 恢复 / 新会话导航）
     4	
     5	注入（stdout additionalContext）：
     6	  1. .omc/session-handoff.md — 会话交接（compact 后恢复）
     7	  2. .omc/state/last-user-prompt.md — 最近用户请求
     8	  3. 活跃 token 状态（task/step/progress）
     9	
    10	设计：只读、快速（<200ms）、永不阻断。无活跃任务时静默退出。
    11	"""
    12	from __future__ import annotations
    13	
    14	import json
    15	import os
    16	import sys
    17	from pathlib import Path
    18	
    19	HOOK_DIR = Path(__file__).resolve().parent
    20	ROOT = HOOK_DIR.parents[1]
    21	os.chdir(str(ROOT))
    22	
    23	OMC = ROOT / ".omc"
    24	HANDOFF = OMC / "session-handoff.md"
    25	LAST_PROMPTS = OMC / "state" / "last-user-prompt.md"
    26	TOKENS_DIR = OMC / "tokens"
    27	
    28	MAX_HANDOFF = 2000
    29	MAX_PROMPTS = 1000
    30	
    31	
    32	def _read_json(path: Path, default):
    33	    try:
    34	        return json.loads(path.read_text(encoding="utf-8"))
    35	    except Exception:
    36	        return default
    37	
    38	
    39	def _active_token_brief() -> str:
    40	    if not TOKENS_DIR.exists():
    41	        return ""
    42	    candidates = sorted(
    43	        [p for p in TOKENS_DIR.glob("*/*.json") if p.is_file()],
    44	        key=lambda p: p.stat().st_mtime, reverse=True,
    45	    )
    46	    for path in candidates[:5]:
    47	        data = _read_json(path, {})
    48	        task = data.get("task")
    49	        if not isinstance(task, dict):
    50	            continue
    51	        stats = data.get("stats", {}) or {}
    52	        session = data.get("session", {}) or {}
    53	        return (
    54	            f"[Active Task] id={path.stem} level={session.get('level', '?')} "
    55	            f"step={task.get('current_step', '?')} done={stats.get('done', 0)}/{stats.get('total', '?')} "
    56	            f"status={task.get('status', data.get('status', '?'))}"
    57	        )
    58	    return ""
    59	
    60	
    61	def main() -> None:
    62	    try:
    63	        payload = json.loads(sys.stdin.read() or "{}")
    64	    except Exception:
    65	        payload = {}
    66	    source = str(payload.get("source") or "startup")
    67	
    68	    parts: list[str] = []
    69	
    70	    token_brief = _active_token_brief()
    71	    if token_brief:
    72	        parts.append(token_brief)
    73	
    74	    if HANDOFF.exists():
    75	        try:
    76	            text = HANDOFF.read_text(encoding="utf-8")[:MAX_HANDOFF]
    77	            if text.strip():
    78	                parts.append(f"[Session Handoff — {source} 恢复导航]\n{text}")
    79	        except Exception:
    80	            pass
    81	
    82	    if source in ("compact", "resume") and LAST_PROMPTS.exists():
    83	        try:
    84	            text = LAST_PROMPTS.read_text(encoding="utf-8")[:MAX_PROMPTS]
    85	            if text.strip():
    86	                parts.append(f"[Last User Prompts]\n{text}")
    87	        except Exception:
    88	            pass
    89	
    90	    if not parts:
    91	        print(json.dumps({"continue": True}))
    92	        sys.exit(0)
    93	
    94	    context = "\n\n".join(parts)
    95	    print(json.dumps({
    96	        "continue": True,
    97	        "hookSpecificOutput": {
    98	            "hookEventName": "SessionStart",
    99	            "additionalContext": context,
   100	        },
   101	    }, ensure_ascii=False))
   102	    sys.exit(0)
   103	
   104	
   105	if __name__ == "__main__":
   106	    main()

===== .claude/references/SUBAGENT.md =====
     1	# SubAgent 派发契约
     2	
     3	> 参考 round3 设计：SubAgent 是轻量任务派发机制，仅在需要时才使用。
     4	
     5	## 何时派发
     6	- 局部调研（读代码/查文档）
     7	- 单一改动（≤5 文件、≤5 分钟）
     8	- 不熟悉的局部技术栈
     9	
    10	**不做**：把整个任务丢给 subagent。要自己做主流程，subagent 只做局部。
    11	
    12	## 派发四步
    13	
    14	1. **写输入**：`subagent/{id}_input.md`
    15	   - Goal（目标）
    16	   - Scope（允许读/改/禁止的范围）
    17	   - 期望产出（什么格式、什么内容）
    18	   - Stop 条件（什么情况下必须停止并返回）
    19	
    20	2. **spawn**：用 delegate_task 工具派发，传 id
    21	
    22	3. **轮询**：每 10s 读 `subagent/{id}.json` 的 status
    23	   - `done` → 合并 deliverable
    24	   - `blocked` → 补信息 / 拆小重派 / 自己接管
    25	   - `15s 无 ACK` 或 `5min 未 done` → 收 partial → 自己接管
    26	
    27	4. **收官**：按状态处理
    28	
    29	## 铁律约束
    30	SubAgent 同样受铁律约束：
    31	- 越界修改 → 违规
    32	- 无证据改动 → 违规
    33	- 埋头不报 > 15s → 违规
    34	
    35	## 契约格式
    36	
    37	```json
    38	{
    39	  "subagent_id": "research-01",
    40	  "goal": "调研 X 模块的 API 设计",
    41	  "scope": {
    42	    "allow_read": ["src/x/**"],
    43	    "allow_write": [],
    44	    "forbidden": ["secrets/**", ".env"]
    45	  },
    46	  "expected_output": "report.md 格式：背景、API 设计、风险点",
    47	  "stop_on": ["timeout_300s", "error_3_consecutive"]
    48	}
    49	```
    50	
    51	## 参考
    52	- poll-subagent.py — 轮询脚本
    53	- subagent-schema.json — 输入/输出 JSON 模式

     1	#!/usr/bin/env python3
     2	"""
     3	handoff_writer.py — Resume Capsule 生成器
     4	
     5	生成 handoff.md（导航-only，非真相源）。
     6	Preflight 验证 token.json CAS + plan 版本一致性。
     7	"""
     8	
     9	import json
    10	from datetime import datetime, timezone
    11	from pathlib import Path
    12	from typing import Optional
    13	
    14	
    15	def generate_capsule(
    16	    task_id: str,
    17	    token: dict,
    18	    plan_path: Optional[Path] = None,
    19	    executor_path: Optional[Path] = None,
    20	    max_nav_chars: int = 8000,
    21	    archived: bool = False,
    22	) -> str:
    23	    """
    24	    生成 Resume Capsule（导航-only）。
    25	    头部强制 NOT_SOURCE_OF_TRUTH。
    26	    如果 archived=True，标记已归档，无 next_action。
    27	    """
    28	    lines = []
    29	    lines.append("# Resume Capsule — Navigation Only")
    30	    lines.append("")
    31	    lines.append("## ⚠️ NOT SOURCE OF TRUTH")
    32	    lines.append("Resume engine MUST load token.json (CAS) first.")
    33	    lines.append("This handoff is navigation only. Do not parse current state from this file.")
    34	    lines.append("")
    35	    lines.append(f"generated: {datetime.now(timezone.utc).isoformat()}")
    36	    lines.append(f"task_id: {task_id}")
    37	    lines.append("")
    38	
    39	    # Goal from token or plan
    40	    lines.append("## Goal")
    41	    if plan_path and plan_path.exists():
    42	        text = plan_path.read_text(encoding="utf-8")
    43	        lines.append(text.splitlines()[0] if text.splitlines() else "?")
    44	    else:
    45	        lines.append(token.get("session", {}).get("id", "?"))
    46	    lines.append("")
    47	
    48	    # Confirmed decisions (from executor if available)
    49	    lines.append("## Confirmed Decisions")
    50	    if executor_path and executor_path.exists():
    51	        for line in executor_path.read_text(encoding="utf-8").splitlines()[-10:]:
    52	            stripped = line.strip()
    53	            if stripped.startswith("-") or stripped.startswith("*"):
    54	                lines.append(f"  {stripped[:160]}")
    55	    if len(lines) < 5:
    56	        lines.append("  (pending)")
    57	    lines.append("")
    58	
    59	    # Next action (or archived)
    60	    if archived:
    61	        lines.append("## ⏹  Status: ARCHIVED")
    62	        lines.append("  This task is archived. Do not resume execution.")
    63	        lines.append("  For reference only. Final report in final-report.md.")
    64	    else:
    65	        lines.append("## Next Action")
    66	        cs = token.get("task", {}).get("current_step", "?")
    67	        st = token.get("task", {}).get("status", "?")
    68	        lines.append(f"  step: {cs} | status: {st}")
    69	    lines.append("")
    70	
    71	    # Changed files from plan scope
    72	    lines.append("## Changed Files")
    73	    if plan_path and plan_path.exists():
    74	        text = plan_path.read_text(encoding="utf-8")
    75	        in_scope = False
    76	        for line in text.splitlines():
    77	            if line.lower().startswith("## scope"):
    78	                in_scope = True
    79	                continue
    80	            if in_scope and line.startswith("## "):
    81	                break
    82	            if in_scope and line.strip().startswith("-"):
    83	                fname = line.strip().lstrip("- ").strip("`")
    84	                if fname:
    85	                    lines.append(f"  - {fname}")
    86	    lines.append("")
    87	
    88	    # Required reads (minimal)
    89	    lines.append("## Required Reads")
    90	    lines.append("  - token.json (CAS)")
    91	    if plan_path:
    92	        lines.append(f"  - {plan_path.relative_to(plan_path.parents[3])}")
    93	    lines.append("")
    94	
    95	    # Do not reload
    96	    lines.append("## Do Not Reload")
    97	    lines.append("  - 全部旧 transcript")
    98	    lines.append("  - docs/reviews/**")
    99	    lines.append("  - 完整测试日志")
   100	    lines.append("  - 旧 Oracle 裁决")
   101	
   102	    text = "\n".join(lines)
   103	    if len(text) > max_nav_chars:
   104	        text = text[: max_nav_chars - 20] + "\n...[TRUNCATED]"
   105	    return text
   106	
   107	
   108	def write_handoff(
   109	    task_dir: Path,
   110	    task_id: str,
   111	    token: dict,
   112	    plan_path: Optional[Path] = None,
   113	    executor_path: Optional[Path] = None,
   114	    archived: bool = False,
   115	) -> Path:
   116	    """Write handoff.md to task directory."""
   117	    capsule = generate_capsule(task_id, token, plan_path, executor_path, archived=archived)
   118	    handoff_path = task_dir / "handoff.md"
   119	    handoff_path.write_text(capsule, encoding="utf-8")
   120	    return handoff_path
   121	
   122	
   123	def run_preflight(task_dir: Path, token: dict) -> list[str]:
   124	    """
   125	    Resume Preflight 检查：
   126	    1) token.json CAS load
   127	    2) plan 版本一致性
   128	    3) external_effects 三界检查
   129	    4) handoff 存在
   130	
   131	    Returns list of issues (empty = pass).
   132	    """
   133	    issues = []
   134	
   135	    # 1. token.json CAS check
   136	    if not token.get("schema_version"):
   137	        issues.append("⚠ token.json missing schema_version (CAS)")
   138	    if not token.get("stats"):
   139	        issues.append("⚠ token.json missing stats section")
   140	
   141	    # 2. plan consistency
   142	    plan_path = task_dir / "plan.md"
   143	    if not plan_path.exists():
   144	        issues.append("⚠ plan.md not found")
   145	    else:
   146	        plan_text = plan_path.read_text(encoding="utf-8")
   147	        if not plan_text.strip():
   148	            issues.append("⚠ plan.md is empty")
   149	
   150	    # 3. external_effects — check state/sub_task dirs for IN_FLIGHT markers
   151	    state_dir = task_dir / "state"
   152	    if state_dir.exists():
   153	        for f in state_dir.iterdir():
   154	            if f.suffix == ".json":
   155	                try:
   156	                    data = json.loads(f.read_text(encoding="utf-8"))
   157	                    effects = data.get("external_effects", []) if isinstance(data, dict) else []
   158	                    for eff in effects:
   159	                        status = eff.get("status", "")
   160	                        if status in ("IN_FLIGHT", "UNKNOWN"):
   161	                            issues.append(f"⚠ external_effect IN_FLIGHT/UNKNOWN: {eff.get('id', '?')}")
   162	                except (json.JSONDecodeError, OSError):
   163	                    pass
   164	
   165	    # 4. handoff exists
   166	    handoff_path = task_dir / "handoff.md"
   167	    if not handoff_path.exists():
   168	        issues.append("ℹ handoff.md not yet written (acceptable for in-progress)")
   169	    else:
   170	        text = handoff_path.read_text(encoding="utf-8")
   171	        if "NOT SOURCE OF TRUTH" not in text:
   172	            issues.append("⚠ handoff.md missing NOT_SOURCE_OF_TRUTH header")
   173	
   174	    return issues

===== .claude/skills/lx-ghost/references/ghost-polling.md =====
     1	# Ghost 全自动轮询
     2	
     3	> 间隔默认 600s，最小 30s，不可为 0。
     4	
     5	## 每轮 Poll 流程
     6	
     7	```
     8	1. 读 lx-ghost.json 确认方向
     9	2. iterations_completed++
    10	3. 方向漂移自检 → 偏离则修正，完全漂移则停用
    11	4. 每轮只做一步 — 不并行 agent，不做大规模分析
    12	5. 危险 → 走裁决链；歧义 → 自主判断
    13	6. 更新状态
    14	```
    15	
    16	## 方向漂移自检
    17	
    18	每轮 poll 开始前检查当前工作是否偏离原始方向。轻微偏离 → 记录 + 修正方向。完全漂移（方向已无交集）→ 触发自动退出。
    19	
    20	## min_iterations 强制探索
    21	
    22	> 解决「一次性做完就停」的过早收敛。
    23	
    24	```
    25	有工作 → 执行一步
    26	无工作 + iterations < min_iterations:
    27	  → 拓宽探索半径 — 扫描 side findings / 边缘问题 / Oracle minor 项 / 相邻文件调用方
    28	  → 仍无 → skip-risk 记录"方向枯竭，强制拓宽"，继续
    29	无工作 + iterations >= min_iterations:
    30	  → 自检: 方向目标是否达成？
    31	  → 是 → 自动退出 + 退出报告
    32	  → 否 → 报告差距，可延长
    33	```
    34	
    35	## 风险与退出
    36	
    37	- 方向漂移 → skip-risk 记录，修正方向
    38	- 修复阻塞（3 次）→ skip-risk，换方向继续
    39	- Context Guard 阻断 → override + Bash
    40	- Permission Gate 拦截 → 走三级裁决链
    41	- 发现范围外问题 → 记入附带发现
    42	- 令牌过高 → 增加间隔或提前关闭
    43	- 硬边界触发 → 立即跳过 → hard-boundary-hit → 继续其他

    57	def _sanitize(text: str) -> str:
    58	    """清除控制字符和代理对"""
    59	    result = []
    60	    for c in text:
    61	        if 0xD800 <= ord(c) <= 0xDFFF:
    62	            continue
    63	        if ord(c) < 0x20 and c not in ("\n", "\t", "\r"):
    64	            continue
    65	        result.append(c)
    66	    return "".join(result)

   411	def cmd_report():
   412	    """生成结构化执行报告"""
   413	    mode_data, path = _read_mode_file()
   414	    report_file = STATE_DIR / "goal-report.md"
   415	
   416	    goal = mode_data.get("goal", "?")
   417	    done = len(mode_data.get("completed_tasks", []))
   418	    skip = len(mode_data.get("skipped_risks", []))
   419	    hard = len(mode_data.get("hard_boundary_hits", []))
   420	    blocked = len(mode_data.get("blocked_human", []))
   421	    retry = mode_data.get("retry_count", 0)
   422	    activated = mode_data.get("activated_at", "?")
   423	    expires = mode_data.get("expires_at", "?")
   424	
   425	    # build lists
   426	    def _skip_line(r):
   427	        if isinstance(r, dict):
   428	            lvl = r.get("risk_level", "low")
   429	            line = f"- [{lvl}] {r.get('description', '?')}"
   430	            if r.get("reason"):
   431	                line += f" — 理由: {r['reason']}"
   432	            if r.get("impact"):
   433	                line += f" / 影响: {r['impact']}"
   434	            return line
   435	        return f"- {r}"
   436	
   437	    skip_list = "\n".join(_skip_line(r) for r in mode_data.get("skipped_risks", [])) or "无"
   438	    hard_list = ""
   439	    for h in mode_data.get("hard_boundary_hits", []):
   440	        hard_list += f"- **操作**: {h.get('description', '?')}\n  **原因**: {h.get('reason', '?')}\n  **需人类执行**: {h.get('human_action', '?')}\n\n"
   441	    hard_list = hard_list or "无"
   442	    blocked_list = ""
   443	    for b in mode_data.get("blocked_human", []):
   444	        blocked_list += f"- **决策**: {b.get('description', '?')}\n  **AI 推荐**: {b.get('ai_recommendation', '?')}\n  **依据**: {b.get('rationale', '?')}\n\n"
   445	    blocked_list = blocked_list or "无"
   446	    task_list = "\n".join(
   447	        f"- [x] {t.get('description', t) if isinstance(t, dict) else t}  ({t.get('timestamp', '') if isinstance(t, dict) else ''})"
   448	        for t in mode_data.get("completed_tasks", [])
   449	    ) or "无"
   450	
   451	    # 人为决策汇总表
   452	    decision_rows = []
   453	    idx = 0
   454	    for h in mode_data.get("hard_boundary_hits", []):
   455	        idx += 1
   456	        decision_rows.append(f"| {idx} | 硬边界 | {h.get('description', '?')} | {h.get('human_action', '?')} | {h.get('reason', '?')} |")
   457	    for b in mode_data.get("blocked_human", []):
   458	        idx += 1
   459	        decision_rows.append(f"| {idx} | 推迟决策 | {b.get('description', '?')} | {b.get('ai_recommendation', '?')} | {b.get('rationale', '?')} |")
   460	    # 中高风险跳过项 — 只跳过不执行，必须反馈人类干预（goal 模式核心安全阀）
   461	    for r in mode_data.get("skipped_risks", []):
   462	        if isinstance(r, dict) and r.get("risk_level") in ("medium", "high", "critical"):
   463	            idx += 1
   464	            basis = r.get("reason") or "风险规避"
   465	            if r.get("impact"):
   466	                basis += f" / 影响: {r['impact']}"
   467	            decision_rows.append(
   468	                f"| {idx} | 中高风险跳过[{r['risk_level']}] | {r.get('description', '?')} "
   469	                f"| 建议人类评估后手动执行或明确放弃 | {basis} |"
   470	            )
   471	    if not decision_rows:
   472	        decision_rows.append("| - | - | 无需人类介入的项 | - | - |")
   473	
   474	    report_content = f"""# 目标模式执行报告
   475	
   476	生成时间: {datetime.now().strftime('%Y-%m-%d %H:%M:%S')}
   477	
   478	## 目标
   479	
   480	{_sanitize(goal)}
   481	
   482	## 基本信息
   483	
   484	- 激活时间: {activated}
   485	- 过期时间: {expires}
   486	
   487	## 执行摘要
   488	
   489	- 已完成任务数: {done}
   490	- 跳过风险数: {skip}
   491	- 硬边界拦截数: {hard}
   492	- 推迟决策数: {blocked}
   493	- 重试次数: {retry}
   494	
   495	## 已完成任务
   496	
   497	{task_list}
   498	
   499	## 跳过的风险
   500	
   501	{skip_list}
   502	
   503	## ⚠️ 需人为决策汇总
   504	
   505	| # | 类型 | 描述 | AI 推荐 | 依据 |
   506	|---|------|------|---------|------|
   507	{chr(10).join(decision_rows)}
   508	
   509	## ⚠️ 需人类介入项（硬边界）
   510	
   511	{hard_list}
   512	
   513	## 推迟决策项（裁决链 Level 3 — 需人类裁决）
   514	
   515	{blocked_list}
   516	
   517	## 验证状态

   520	"""
   521	    report_file.write_text(report_content, encoding="utf-8")
   522	    print(f"✅ 报告已生成: {report_file}")
   523	    print(report_content)

   550	                    print("   生成过期报告...")

   552	                MODE_FILE.unlink(missing_ok=True)
   553	                AUTONOMOUS_SIGNAL.unlink(missing_ok=True)
   554	                return

   556	            pass

   558	    goal = data.get("goal", "?")
   559	    done = len(data.get("completed_tasks", []))
   560	    skip = len(data.get("skipped_risks", []))

   676	    """记录 subagent 分配/回收/结果"""
   677	    if not MODE_FILE.exists():
   678	        print("❌ 目标模式未开启")
   679	        sys.exit(1)
   680	    mode_data, _ = _read_mode_file()

   694	        elif action == "fail":
   695	            f.write(f"\n### {ts} — 失败: {agent_name} → {subtask}\n  - 原因: {detail}\n")
   696	        elif action == "summary":
   697	            f.write(f"\n---\n### SubAgent 执行摘要 @ {ts}\n")

   701	            f.write(f"- 完成: {content.count('完成:')}\n")
   702	            f.write(f"- 失败: {content.count('失败:')}\n")
   703	        else:
   704	            print("用法: lx-goal.py subagent-log assign|complete|fail|summary <agent> <subtask> [detail]")
   705	            sys.exit(1)
   706	
   707	    print(f"📝 subagent 日志已更新: {executor_md}")

   727	        print(f"🔓 物理锁已删除: {lock_file}")

   730	        sf = plan_dir / "state.json"
   731	        if sf.exists():
   732	            with open(sf, encoding="utf-8") as f:
   733	                sd = json.load(f)

   735	            sd["accepted_at"] = get_now()

   740	
   741	    # 更新 plan.md
