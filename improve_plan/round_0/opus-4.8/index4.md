### 二、Resume 引擎完整算法

```python
from dataclasses import dataclass
from enum import Enum

class ResumeStatus(Enum):
    OK = "ok"
    BLOCKED = "blocked"
    UNSAFE = "unsafe"
    INCONSISTENT = "inconsistent"

@dataclass
class ResumeResult:
    status: ResumeStatus
    capsule: Optional[ContextCapsule]
    resume_prompt: Optional[str]
    blockers: List[str]
    warnings: List[str]
    required_actions: List[str]


def resume_task(task_id: str, session_id: str) -> ResumeResult:
    """
    完整 Resume 引擎
    按照确定性顺序从磁盘重建状态
    """
    
    blockers = []
    warnings = []
    required_actions = []
    
    # === Phase 1: 加载核心状态 ===
    try:
        state = load_token(task_id)
        plan = load_plan(task_id)
        manifest = load_manifest(task_id)
    except FileNotFoundError as e:
        return ResumeResult(
            status=ResumeStatus.BLOCKED,
            capsule=None,
            resume_prompt=None,
            blockers=[f"Missing core file: {e.filename}"],
            warnings=[],
            required_actions=["Restore from backup or recreate task"]
        )
    
    # === Phase 2: 一致性检查 ===
    
    # 2.1 版本一致性
    if state.plan_version != plan.version:
        blockers.append(
            f"State references plan v{state.plan_version}, "
            f"but plan is v{plan.version}"
        )
    
    if state.manifest_version != manifest.version:
        blockers.append(
            f"State references manifest v{state.manifest_version}, "
            f"but manifest is v{manifest.version}"
        )
    
    # 2.2 步骤存在性
    if state.current_step not in plan.steps:
        blockers.append(
            f"State references step {state.current_step}, "
            f"but plan does not define it"
        )
    
    # 2.3 状态合法性
    if state.status == "ARCHIVED":
        blockers.append("Task is archived, cannot resume")
    
    if state.status == "CANCELLED":
        warnings.append("Task was cancelled, resume will restart it")
    
    # === Phase 3: 外部副作用检查 ===
    
    unsafe_effects = []
    for effect in state.external_effects:
        if effect.status in ["IN_FLIGHT", "UNKNOWN"]:
            unsafe_effects.append(effect)
    
    if unsafe_effects:
        blockers.append(
            f"Unsafe external effects detected: "
            f"{[e.effect_id for e in unsafe_effects]}"
        )
        required_actions.append(
            "Human must verify status of external effects:\n" +
            "\n".join([
                f"  - {e.effect_id}: {e.description} (status: {e.status})"
                for e in unsafe_effects
            ])
        )
    
    # === Phase 4: Checkpoint 一致性 ===
    
    if state.last_checkpoint_id:
        checkpoint = load_checkpoint(state.last_checkpoint_id)
        if checkpoint.state_version != state.version:
            warnings.append(
                f"Last checkpoint is for state v{checkpoint.state_version}, "
                f"current state is v{state.version}. "
                f"Rollback may not be accurate."
            )
    
    # === Phase 5: 决定是否可以 Resume ===
    
    if blockers:
        return ResumeResult(
            status=ResumeStatus.BLOCKED,
            capsule=None,
            resume_prompt=None,
            blockers=blockers,
            warnings=warnings,
            required_actions=required_actions
        )
    
    # === Phase 6: 重建 Context Capsule ===
    
    try:
        capsule = build_context_capsule(
            task_id=task_id,
            turn=state.turns + 1
        )
    except Exception as e:
        return ResumeResult(
            status=ResumeStatus.BLOCKED,
            capsule=None,
            resume_prompt=None,
            blockers=[f"Failed to build context capsule: {str(e)}"],
            warnings=warnings,
            required_actions=["Check working-set.yaml and referenced files"]
        )
    
    # === Phase 7: 生成 Resume Prompt ===
    
    # 读取 handoff（仅作为导航）
    handoff_exists = os.path.exists(f".omc/task/{task_id}/handoff.md")
    handoff_context = ""
    if handoff_exists:
        handoff = load_handoff(task_id)
        suggested_files = extract_suggested_files(handoff)
        stop_reason = extract_stop_reason(handoff)
        
        handoff_context = f"""
## Context from Previous Session

**Why session stopped**: {stop_reason}

**Suggested files to review**:
{format_file_list(suggested_files)}
"""
    
    resume_prompt = f"""
# Resuming Task: {manifest.goal}

⚠️ **CRITICAL**: This session is resuming after context was cleared.

**DO NOT rely on conversation memory.**
All state has been reconstructed from persistent files.

---

## Task State (from token.json)

- **Task ID**: {state.task_id}
- **Current Step**: {state.current_step}
- **Status**: {state.current_step_status}
- **Last Verified Step**: {state.verification.last_verified_step}
- **Progress**: {state.progress.completed_steps}/{state.progress.total_steps} steps

## Active Blocker

{state.blocker or "None"}

## Next Action

{state.next_action or "Continue current step"}

{handoff_context}

---

## Your Instructions

1. Review the Hot Tail below for current state
2. Review suggested files (if any)
3. Continue from current step: {state.current_step}
4. **Do NOT**:
   - Skip ahead to next step
   - Assume previous work is verified (check evidence)
   - Re-do verified steps
   - Change the goal

Ready to proceed?

---

{capsule.tail}
"""
    
    # === Phase 8: 更新状态 ===
    
    update_state(
        task_id=task_id,
        updates={
            "last_resumed_at": now(),
            "last_resumed_by": session_id,
            "turns": state.turns + 1,
        },
        expected_version=state.version
    )
    
    return ResumeResult(
        status=ResumeStatus.OK,
        capsule=capsule,
        resume_prompt=resume_prompt,
        blockers=[],
        warnings=warnings,
        required_actions=[]
    )
```

### 三、外部副作用对账机制

```python
from enum import Enum

class EffectStatus(Enum):
    PENDING = "pending"        # 计划执行但未执行
    IN_FLIGHT = "in_flight"    # 正在执行
    COMMITTED = "committed"    # 已完成且不可逆
    ROLLED_BACK = "rolled_back"  # 已回滚
    UNKNOWN = "unknown"        # 状态未知（危险）

@dataclass
class ExternalEffect:
    effect_id: str
    kind: str  # git_commit, api_call, file_write, db_mutation
    description: str
    status: EffectStatus
    reversible: bool
    checkpoint_id: Optional[str]
    metadata: dict


def reconcile_external_effects(
    task_id: str,
    checkpoint_id: Optional[str] = None
) -> ReconcileResult:
    """
    对账外部副作用
    确保状态文件与实际世界一致
    """
    
    state = load_token(task_id)
    mismatches = []
    
    for effect in state.external_effects:
        
        # === Git Commit 检查 ===
        if effect.kind == "git_commit":
            commit_exists = check_git_commit_exists(effect.metadata["sha"])
            
            if effect.status == "COMMITTED" and not commit_exists:
                mismatches.append({
                    "effect_id": effect.effect_id,
                    "expected": "COMMITTED",
                    "actual": "NOT_FOUND",
                    "action": "Mark as UNKNOWN or rollback state"
                })
            
            if effect.status == "PENDING" and commit_exists:
                mismatches.append({
                    "effect_id": effect.effect_id,
                    "expected": "PENDING",
                    "actual": "COMMITTED",
                    "action": "Update state to COMMITTED"
                })
        
        # === File Write 检查 ===
        elif effect.kind == "file_write":
            file_path = effect.metadata["path"]
            expected_hash = effect.metadata.get("content_hash")
            
            if not os.path.exists(file_path):
                if effect.status == "COMMITTED":
                    mismatches.append({
                        "effect_id": effect.effect_id,
                        "expected": "file exists",
                        "actual": "file missing",
                        "action": "Restore from checkpoint or mark UNKNOWN"
                    })
            else:
                actual_hash = compute_file_hash(file_path)
                if expected_hash and actual_hash != expected_hash:
                    mismatches.append({
                        "effect_id": effect.effect_id,
                        "expected": f"hash {expected_hash[:8]}",
                        "actual": f"hash {actual_hash[:8]}",
                        "action": "File modified after commit, verify safety"
                    })
        
        # === API Call 检查（通常不可对账）===
        elif effect.kind == "api_call":
            if effect.status == "IN_FLIGHT":
                # 无法确定是否完成
                mismatches.append({
                    "effect_id": effect.effect_id,
                    "expected": "completed or failed",
                    "actual": "UNKNOWN",
                    "action": "Human must verify API call status"
                })
    
    return ReconcileResult(
        consistent=(len(mismatches) == 0),
        mismatches=mismatches
    )


def mark_effect_committed(
    task_id: str,
    effect_id: str,
    metadata: dict = None
):
    """标记副作用已完成"""
    state = load_token(task_id)
    
    for effect in state.external_effects:
        if effect.effect_id == effect_id:
            effect.status = EffectStatus.COMMITTED
            if metadata:
                effect.metadata.update(metadata)
            break
    
    update_state(
        task_id,
        {"external_effects": state.external_effects},
        expected_version=state.version
    )
```

**关键原则**：

1. **Git Commit 可验证**：检查 commit SHA 是否存在
2. **File Write 可哈希**：对比文件内容哈希
3. **API Call 不可对账**：必须标记为 UNKNOWN，要求人工确认
4. **Resume 前必须对账**：检测到 mismatch 就阻止 Resume

### 四、Checkpoint 与 Resume 的协作关系

```python
@dataclass
class Checkpoint:
    checkpoint_id: str
    task_id: str
    created_at: str
    state_version: int
    
    # 快照内容
    state_snapshot: dict
    plan_snapshot: dict
    working_files_snapshot: Dict[str, str]  # path -> content_hash
    
    # 外部副作用状态
    committed_effects: List[str]  # effect_ids
    reversible_effects: List[str]
    
    # Git 状态
    git_ref: Optional[str]
    git_uncommitted_files: List[str]


def create_checkpoint(task_id: str, reason: str) -> Checkpoint:
    """创建 checkpoint"""
    state = load_token(task_id)
    plan = load_plan(task_id)
    
    # 快照工作文件
    working_set = load_working_set(task_id)
    files_snapshot = {}
    for doc_ref in working_set.docs:
        if doc_ref.id.startswith("src/"):
            content = read_file(doc_ref.id)
            files_snapshot[doc_ref.id] = hashlib.sha256(content.encode()).hexdigest()
    
    # Git 状态
    git_ref = run_command("git rev-parse HEAD").strip()
    git_status = run_command("git status --porcelain")
    uncommitted = [
        line.split()[1]
        for line in git_status.splitlines()
    ]
    
    checkpoint = Checkpoint(
        checkpoint_id=f"CHK-{task_id}-{state.version}",
        task_id=task_id,
        created_at=now(),
        state_version=state.version,
        state_snapshot=state,
        plan_snapshot=plan,
        working_files_snapshot=files_snapshot,
        committed_effects=[
            e.effect_id
            for e in state.external_effects
            if e.status == EffectStatus.COMMITTED
        ],
        reversible_effects=[
            e.effect_id
            for e in state.external_effects
            if e.reversible
        ],
        git_ref=git_ref,
        git_uncommitted_files=uncommitted,
    )
    
    # 持久化
    write_checkpoint(checkpoint)
    
    # 更新状态
    update_state(
        task_id,
        {"last_checkpoint_id": checkpoint.checkpoint_id},
        expected_version=state.version
    )
    
    return checkpoint


def rollback_to_checkpoint(checkpoint_id: str) -> RollbackResult:
    """回滚到 checkpoint"""
    checkpoint = load_checkpoint(checkpoint_id)
    current_state = load_token(checkpoint.task_id)
    
    # === 检查可回滚性 ===
    
    # 1. 检查新产生的不可逆副作用
    new_committed_effects = [
        e
        for e in current_state.external_effects
        if e.status == EffectStatus.COMMITTED
        and e.effect_id not in checkpoint.committed_effects
        and not e.reversible
    ]
    
    if new_committed_effects:
        return RollbackResult(
            success=False,
            reason="New irreversible effects exist",
            effects=new_committed_effects,
            action="Cannot rollback, must continue forward"
        )
    
    # 2. 检查 Git 状态
    current_git_ref = run_command("git rev-parse HEAD").strip()
    if current_git_ref != checkpoint.git_ref:
        # 有新的 commit
        return RollbackResult(
            success=False,
            reason=f"Git moved from {checkpoint.git_ref[:8]} to {current_git_ref[:8]}",
            action="Use git reset or continue forward"
        )
    
    # === 执行回滚 ===
    
    # 恢复状态文件
    write_token(checkpoint.task_id, checkpoint.state_snapshot)
    write_plan(checkpoint.task_id, checkpoint.plan_snapshot)
    
    # 恢复工作文件（如果哈希匹配）
    for path, expected_hash in checkpoint.working_files_snapshot.items():
        if os.path.exists(path):
            current_hash = compute_file_hash(path)
            if current_hash != expected_hash:
                # 文件已修改，需要决策
                print(f"Warning: {path} was modified, skipping restore")
                continue
        
        # 从 checkpoint 恢复
        restore_file_from_checkpoint(checkpoint_id, path)
    
    return RollbackResult(
        success=True,
        reason="Rolled back successfully",
        restored_files=list(checkpoint.working_files_snapshot.keys())
    )
```

**Checkpoint 策略**：

```text
何时创建 checkpoint：
✓ 进入 L2 工作流
✓ 执行不可逆操作之前
✓ 用户明确要求
✓ 长时间运行任务的定期备份

何时不需要 checkpoint：
✗ L1 快速工作流（Git 已经足够）
✗ 只读操作
✗ 频繁小改动（成本高）
```

### 五、Compact 后禁止跳过 VerifyGate

```python
def detect_compact_bypass(task_id: str) -> BypassDetection:
    """
    检测 compact 后是否绕过了 VerifyGate
    """
    
    state = load_token(task_id)
    
    # 检查最近是否经历了 compact
    if state.platform.compaction_generation == 0:
        return BypassDetection(detected=False)
    
    # 检查 compact 后是否直接修改了 step status
    operation_log = load_operation_log(task_id)
    
    last_compact_time = None
    for entry in reversed(operation_log):
        if entry.action == "compaction_detected":
            last_compact_time = entry.timestamp
            break
    
    if not last_compact_time:
        return BypassDetection(detected=False)
    
    # 检查 compact 后是否有未经验证的 step 变更
    suspicious_changes = []
    for entry in operation_log:
        if entry.timestamp <= last_compact_time:
            continue
        
        if entry.action == "step_marked_verified":
            # 检查是否有对应的 verdict
            verdict_exists = check_verdict_exists(
                task_id,
                entry.metadata["step_id"]
            )
            if not verdict_exists:
                suspicious_changes.append({
                    "timestamp": entry.timestamp,
                    "step": entry.metadata["step_id"],
                    "reason": "marked VERIFIED without verdict"
                })
        
        if entry.action == "task_archived":
            # 检查所有 step 是否都有 verdict
            all_verified = all(
                check_verdict_exists(task_id, step_id)
                for step_id in state.plan.steps.keys()
            )
            if not all_verified:
                suspicious_changes.append({
                    "timestamp": entry.timestamp,
                    "reason": "archived without all steps verified"
                })
    
    return BypassDetection(
        detected=(len(suspicious_changes) > 0),
        violations=suspicious_changes
    )
```

**硬约束**：

```text
compact 后禁止：
✗ 凭记忆声称 step 已完成
✗ 修改 plan.md 的 VERIFIED 标记
✗ 绕过 VerifyGate 直接进入下一步
✗ 声称"之前已经验证过"

compact 后必须：
✓ Resume 引擎重建状态
✓ 重新读取 evidence.jsonl
✓ 如果 step 未验证，重新执行验证
✓ 记录 operation log
```

def test_resume_blocks_on_unknown_external_effect():
    """未知外部副作用阻止 Resume"""
    task_id = "test-001"
    
    # 创建有 UNKNOWN effect 的状态
    state = {
        "task_id": task_id,
        "status": "RUNNING",
        "external_effects": [
            {
                "effect_id": "EXT-001",
                "kind": "api_call",
                "status": "UNKNOWN",
                "reversible": False,
            }
        ],
    }
    write_token(task_id, state)
    
    # Resume 必须被阻止
    result = resume_task(task_id, "session-02")
    assert result.status == ResumeStatus.BLOCKED
    assert "unsafe external effects" in result.blockers[0].lower()


def test_checkpoint_prevents_irreversible_rollback():
    """不可逆副作用阻止回滚"""
    task_id = "test-002"
    
    # 创建 checkpoint
    checkpoint = create_checkpoint(task_id, "before API call")
    
    # 执行不可逆操作

        "effect_id": "EXT-002",
        "kind": "api_call",
        "status": "COMMITTED",
        "reversible": False,
    })
    
    # 尝试回滚
    result = rollback_to_checkpoint(checkpoint.checkpoint_id)
    assert result.success is False
    assert "irreversible effects" in result.reason.lower()
